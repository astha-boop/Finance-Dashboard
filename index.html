// ── Sheet 1: Day-to-Day P&L ──────────────────────────────────────
// Column layout (0-indexed):
//   A(0) = Date          B(1) = Expense desc   C(2) = Revenue desc
//   D(3) = Debit         E(4) = Credit          F(5) = Net
//   G(6) = Bank          H(7) = Expense Cat     I(8) = Revenue Cat
// First 3 rows are headers → data starts at row index 3

const TOKEN = "pradeep2026"; // ← must match token in dashboard

function doGet(e)  { return run(e.parameter, null); }
function doPost(e) {
  var b = {};
  try { b = JSON.parse(e.postData.contents || "{}"); } catch(x) {}
  return run(e.parameter, b);
}

function run(params, body) {
  var p = body || params || {};
  if (p.token !== TOKEN) return out({ok:false, error:"Unauthorized"});
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  if (p.action === "test") {
    return out({ok:true, title:ss.getName(),
      sheets: ss.getSheets().map(function(s){ return s.getName(); })});
  }

  if (p.action === "pull") {
    var tabs = p.tabs || {
      jan: "January - 2026",
      feb: "Feb-2026",
      mar: "March-2026"
    };
    var result = {};
    for (var key in tabs) {
      var sheet = ss.getSheetByName(tabs[key]);
      if (!sheet) { result[key] = []; continue; }
      var rows = sheet.getDataRange().getValues();
      var data = [];
      for (var i = 3; i < rows.length; i++) {   // skip 3 header rows
        var r = rows[i];
        var date    = String(r[0] || "").trim();
        var expDesc = String(r[1] || "").trim();  // col B: expense description
        var revDesc = String(r[2] || "").trim();  // col C: revenue description
        var debit   = parseFloat(r[3]) || 0;      // col D: debit amount
        var credit  = parseFloat(r[4]) || 0;      // col E: credit amount
        var bank    = String(r[6] || "").trim();   // col G: bank name
        var expCat  = String(r[7] || "").trim();   // col H: expense category
        var revCat  = String(r[8] || "").trim();   // col I: revenue category
        if (!date && !expDesc && !revDesc && debit === 0 && credit === 0) continue;
        data.push({
          date:   date,
          desc:   expDesc || revDesc,
          cat:    expCat  || revCat || "Other",
          bank:   bank    || "Kotak",
          debit:  debit,
          credit: credit
        });
      }
      result[key] = data;
    }
    return out({ok:true, action:"pull", data:result});
  }

  return out({ok:false, error:"Unknown action: " + p.action});
}

function out(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

// ── Sheet 2: Q4 Sales Sheet ──────────────────────────────────────
// ALL tabs are learner registration sheets with this layout (0-indexed):
//   A(0)  = Timestamp        B(1) = First Name    C(2) = Last Name
//   D(3)  = Mobile           E(4) = Alt Mobile    F(5) = Email
//   G(6)  = [gap column]
//   H(7)  = Total Fee        ← SALES VALUE per learner
//   I(8)  = Reg Amount       ← Upfront amount paid
//   J(9)  = Date of Registration
//   K(10) = Counsellor Name  ← team attribution
//   ...
//   Z(25) = Amount Received  ← actual cash collected
// Row 0 = header → data starts row 1

const TOKEN = "pradeep2026"; // ← same token as Sheet 1

function doGet(e)  { return run(e.parameter, null); }
function doPost(e) {
  var b = {};
  try { b = JSON.parse(e.postData.contents || "{}"); } catch(x) {}
  return run(e.parameter, b);
}

function run(params, body) {
  var p = body || params || {};
  if (p.token !== TOKEN) return out({ok:false, error:"Unauthorized"});
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  if (p.action === "test") {
    return out({ok:true, title:ss.getName(),
      sheets: ss.getSheets().map(function(s){ return s.getName(); })});
  }

  if (p.action === "pullSales") {
    // Tab groups per month — reads all tabs and merges
    var monthTabs = p.monthTabs || {
      jan: ["Jan", "January Sales"],
      feb: ["feb(A)", "February Sales"],
      mar: ["March Sales Rev. Based", "Mar Sales", "Activation Course- March"]
    };

    var result = {};

    for (var month in monthTabs) {
      var tabNames = monthTabs[month];
      var allLearners = [];

      for (var t = 0; t < tabNames.length; t++) {
        var sheet = ss.getSheetByName(tabNames[t]);
        if (!sheet) continue;
        var rows = sheet.getDataRange().getValues();

        for (var i = 1; i < rows.length; i++) {   // row 0 = header
          var r = rows[i];
          var timestamp = String(r[0] || "").trim();
          if (!timestamp) continue;                 // skip empty rows

          var fee        = parseFloat(String(r[7]  || "").replace(/[^0-9.]/g,"")) || 0;  // col H: Total Fee
          var regAmt     = parseFloat(String(r[8]  || "").replace(/[^0-9.]/g,"")) || 0;  // col I: Reg Amount
          var regDate    = String(r[9]  || "").trim();  // col J
          var counsellor = String(r[10] || "").trim();  // col K
          var firstName  = String(r[1]  || "").trim();  // col B
          var lastName   = String(r[2]  || "").trim();  // col C
          var received   = parseFloat(String(r[25] || "").replace(/[^0-9.]/g,"")) || 0;  // col Z

          if (fee > 0 || received > 0) {
            allLearners.push({
              name:       (firstName + " " + lastName).trim(),
              counsellor: counsellor || "Unassigned",
              regDate:    regDate,
              fee:        fee,
              regAmt:     regAmt,
              received:   received,
              tab:        tabNames[t]
            });
          }
        }
      }

      result[month] = {
        totalFee:      allLearners.reduce(function(s,l){ return s + l.fee;      }, 0),
        totalReceived: allLearners.reduce(function(s,l){ return s + l.received; }, 0),
        count:         allLearners.length,
        learners:      allLearners
      };
    }

    return out({ok:true, action:"pullSales", data:result});
  }

  return out({ok:false, error:"Unknown action: " + p.action});
}

function out(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
